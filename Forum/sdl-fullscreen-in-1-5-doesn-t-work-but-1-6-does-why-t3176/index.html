What could be the difference in 1.6 and 1.5 that makes SDL work in fullscreen with 1.6, but not work with 1.5?

I took the exact same SDL calls and code:

Code: Select all
  1. //----------------------------------------------------------------//

  2. void _AKUEnterFullscreenModeFunc () {

  3.  

  4.     //videomode change

  5.     SDL_SetWindowFullscreen(sWindow, SDL_WINDOW_FULLSCREEN);

  6.     SetScreenSize( SDL_GetCurrentDisplayMode ); // copied again from 1.6, etc

But the fullscreen goes totally bad with 1.5, going to quarter of the screen, corruption showing other windows, or taking up all monitors on multimonitor, etc.

Re: SDL Fullscreen in 1.5 doesn't work, but 1.6 does - why?

by naturally » Thu Jul 16, 2015 8:40 pm

Think I may have figured it out. SDL isn't compiled with X11 stuff in 1.5.

Need these libs:
sudo apt-get install libxrandr-dev libx11-dev libxinerama-dev

Change host-sdl cmake (not sure if required):
Code: Select all
  1. cmake_minimum_required(VERSION 2.8.5)

  2.  

  3. project(host-sdl)

  4.  

  5. if ( BUILD_OSX )

  6.   find_library ( APPLICATION_SERVICES_LIBRARY ApplicationServices )

  7.   find_library ( CORE_SERVICES_LIBRARY CoreServices )

  8.   find_package ( OpenGL REQUIRED )

  9. endif ( BUILD_OSX )

  10.  

  11. set ( INCLUDES_MOAIHOSTSDL

  12.   "${MOAI_ROOT}/src/"

  13.   "${MOAI_ROOT}/src/config-default/"

  14.   "${MOAI_ROOT}/src/host-sdl/"

  15.   "${MOAI_ROOT}/3rdparty/sdl2-2.0.0/include"

  16.    ${OPENGL_INCLUDE_DIRS}

  17. )

  18.  

  19. find_package(X11)

  20.    

  21. FIND_LIBRARY(X11_Xxf86vm_LIB Xxf86vm ${X11_LIB_SEARCH_PATH})

  22. FIND_LIBRARY(XI_LIBRARY Xi ${X11_LIB_SEARCH_PATH})

  23. MARK_AS_ADVANCED(X11_Xxf86vm_LIB XI_LIBRARY)

  24.  

  25. include_directories ( ${INCLUDES_MOAIHOSTSDL} )

  26.  

  27. set ( CMAKE_CURRENT_SOURCE_DIR "${MOAI_ROOT}/src/host-sdl" )

  28.  

  29. # Use all cpp files in directory as source code

  30. file ( GLOB HOST_SDL_SRC "${CMAKE_CURRENT_SOURCE_DIR}/*.cpp" "${CMAKE_CURRENT_SOURCE_DIR}/*.h" )

  31.  

  32. add_executable ( moai ${HOST_SDL_SRC} )

  33.  

  34. target_link_libraries ( moai

  35.   host-modules

  36.   moai-sim

  37.   moai-util

  38.   moai-core

  39.   zlcore

  40.   SDL2-static

  41. )

  42.  

  43. if ( BUILD_WINDOWS )

  44.   target_link_libraries ( moai

  45.     SDL2-static

  46.     ${OPENGL_LIBRARIES}

  47.     winmm

  48.   )

  49. endif (BUILD_WINDOWS)

  50.  

  51. if ( BUILD_LINUX )

  52.   target_link_libraries ( moai

  53.     pthread

  54.     rt

  55.     Xxf86vm

  56.         ${XI_LIBRARY}

  57.         ${OPENGL_LIBRARIES}

  58.         ${X11_Xxf86vm_LIB}

  59.         ${XCB_XCB_LIBRARIES}

  60.         ${XCB_RANDR_LIBRARIES}

  61.         ${XCB_UTIL_LIBRARIES}

  62.   )

  63. endif ( BUILD_LINUX )

  64.  

  65. if ( BUILD_OSX )

  66.   target_link_libraries ( moai

  67.     ${APPLICATION_SERVICES_LIBRARY}

  68.     ${CORE_SERVICES_LIBRARY}

  69.     ${OPENGL_LIBRARIES}

  70.   )

  71.  

  72.   macro(ADD_FRAMEWORK fwname appname)

  73.     find_library(FRAMEWORK_${fwname}

  74.         NAMES ${fwname}

  75.         PATHS ${CMAKE_SYSTEM_FRAMEWORK_PATH}

  76.         NO_DEFAULT_PATH)

  77.     if( ${FRAMEWORK_${fwname}} STREQUAL FRAMEWORK_${fwname}-NOTFOUND)

  78.         MESSAGE(ERROR ": Framework ${fwname} not found")

  79.     else()

  80.         TARGET_LINK_LIBRARIES(${appname} "${FRAMEWORK_${fwname}}/${fwname}")

  81.         MESSAGE(STATUS "Framework ${fwname} found at ${FRAMEWORK_${fwname}}")

  82.     endif()

  83.   endmacro(ADD_FRAMEWORK)

  84.  

  85.   macro(ADD_FRAMEWORKS appname fwnames)

  86.     FOREACH(FW ${fwnames})

  87.       ADD_FRAMEWORK(${FW} ${appname})

  88.     ENDFOREACH(FW)

  89.   endmacro(ADD_FRAMEWORKS)

  90.  

  91.   set( FWORKS

  92.     AudioToolbox

  93.     AudioUnit

  94.     CoreAudio

  95.     CoreGraphics

  96.     CoreLocation

  97.     Foundation

  98.     GameKit

  99.     QuartzCore

  100.     StoreKit

  101.     SystemConfiguration

  102.   )

  103.  

  104.   ADD_FRAMEWORKS(moai "${FWORKS}")

  105.  

  106.   if (MOAI_LUAJIT)

  107.      set (CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -pagezero_size 10000 -image_base 100000000 ")

  108.   endif (MOAI_LUAJIT)

  109.  

  110. endif ( BUILD_OSX )
