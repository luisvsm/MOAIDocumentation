  1. // Copyright (c) 2010-2011 Zipline Games, Inc. All Rights Reserved.

  2. // <!-- m --><a class="postlink" href=""></a><!-- m -->

  3.  

  4. #include "pch.h"

  5. #include <moai-sim/MOAIJoystickSensor.h>

  7.  

  8. //================================================================//

  9. // MOAIJoystickEvent

  10. //================================================================//

  11. class MOAIJoystickEvent {

  12. public:

  13.         u8              mA;

  14.         float   mX;

  15.         float   mY;

  16. };

  17.  

  18. //================================================================//

  19. // lua

  20. //================================================================//

  21.  

  22. //----------------------------------------------------------------//

  23. /**     @lua    getVector

  24.         @text   Returns the joystick vector.

  25.  

  26.         @in             MOAIJoystickSensor self

  27.         @out    number x

  28.         @out    number y

  29. */

  30. int MOAIJoystickSensor::_getVector ( lua_State* L ) {

  31.         MOAI_LUA_SETUP ( MOAIJoystickSensor, "U" )

  32.        

  33.         lua_pushnumber ( state, self->mX );

  34.         lua_pushnumber ( state, self->mY );

  35.         return 2;

  36. }

  37.  

  38. //----------------------------------------------------------------//

  39. /**     @lua    setCallback

  40.         @text   Sets or clears the callback to be issued when the joystick vector changes.

  41.  

  42.         @in             MOAIJoystickSensor self

  43.         @opt    function callback                       Default value is nil.

  44.         @out    nil

  45. */

  46. int MOAIJoystickSensor::_setCallback ( lua_State* L ) {

  47.         MOAI_LUA_SETUP ( MOAIJoystickSensor, "U" )

  48.        

  49.         self->mOnStick.SetRef ( state, 3 );

  50.        

  51.         return 0;

  52. }

  53.

  55. // MOAIJoystickSensor

  59. void MOAIJoystickSensor::EnqueueJoystickEvent( MOAIInputQueue& queue, u8 deviceID, u8 sensorID, u8 stateID, float x, float y ) {

  60.  

  61.         if ( queue.WriteEventHeader < MOAIJoystickSensor >( deviceID, sensorID )) {

  62.                 queue.Write < u8 >( stateID );

  63.                 queue.Write < float >( x );

  64.                 queue.Write < float >( y );

  65.         }

  66. }

  67.

  76. }

  78. //----------------------------------------------------------------//

  79. void MOAIJoystickSensor::ParseEvent ( ZLStream& eventStream ) {

  80.  

  81.         this->mA = eventStream.Read < u8 >( 1 );

  82.         this->mX = eventStream.Read < float >( 0.0f );

  83.         this->mY = eventStream.Read < float >( 0.0f );

  84.        

  85.         if ( this->mOnStick ) {

  86.                 MOAIScopedLuaState state = this->mOnStick.GetSelf ();

  87.                 lua_pushnumber ( state, this->mA );

  88.                 lua_pushnumber ( state, this->mX );

  89.                 lua_pushnumber ( state, this->mY );

  90.                 state.DebugCall ( 3, 0 );

  91.         }

  92. }

by naturally » Wed Apr 15, 2015 2:31 pm

Have a solution:

Code: Select all
  1. case SDL_JOYBUTTONDOWN:

  2. case SDL_JOYBUTTONUP:

  3.         AKUEnqueueJoystickEvent ( InputDeviceID::DEVICE, InputSensorID::JOYSTICK, sdlEvent.jbutton.button+2, (sdlEvent.jbutton.state == SDL_PRESSED) );

  4.  



Just add 2 to the call because axis movement is -1 to 1. Then test if >=2 in Lua.

Hacky but it works.
