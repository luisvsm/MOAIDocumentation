  296. @Override

  297.     public boolean onGenericMotionEvent(MotionEvent event) {

  298.         boolean handled = OuyaController.onGenericMotionEvent(event);

  299.      

  300.         int player = OuyaController.getPlayerNumByDeviceId(event.getDeviceId());

  301.  

  302.  

  303.    

  304.         if ((event.getSource() & InputDevice.SOURCE_CLASS_POINTER) != 0){

  305.  

  306.                 float touchpadX = event.getX();

  307.                 float touchpadY = event.getY();

  308.  

  309.               //  MoaiOuya.NotifyOuyaMotionEventTouchpad( touchpadX, touchpadY, player);

  310.         }

  311.  

  312.         else{

  313.                 float leftAxisX = event.getAxisValue(OuyaController.AXIS_LS_X);

  314.                 float leftAxisY = event.getAxisValue(OuyaController.AXIS_LS_Y);

  315.                 float rightAxisX = event.getAxisValue(OuyaController.AXIS_RS_X);

  316.                 float rightAxisY = event.getAxisValue(OuyaController.AXIS_RS_Y);

  317.                 float l2Axis = event.getAxisValue(OuyaController.AXIS_L2);

  318.                 float r2Axis = event.getAxisValue(OuyaController.AXIS_R2);

  319.  

  320.                 boolean callNotification = false;

  321.                 float c_minStickDistance = OuyaController.STICK_DEADZONE * OuyaController.STICK_DEADZONE;

  322.  

  323.                 if (leftAxisX * leftAxisX + leftAxisY * leftAxisY < c_minStickDistance){

  324.                     leftAxisX = leftAxisY = 0.0f;

  325.                 }

  326.                 else{

  327.                        

  328.                     callNotification = true;

  329.                 }

  330.  

  331.                 if (rightAxisX * rightAxisX + rightAxisY * rightAxisY < c_minStickDistance){

  332.                     rightAxisX = rightAxisY = 0.0f;

  333.                     callNotification = true;

  334.                 }

  335.                 else{

  336.                     callNotification = true;

  337.                 }

  338.  

  339.                 if (l2Axis > 0.0f || r2Axis > 0.0f ){

  340.                     callNotification = true;

  341.                 }

  342.                

  343.                 if ( callNotification ){

  344.                        

  345.                                         Moai.buttonMotion(leftAxisX, leftAxisY, rightAxisX, rightAxisY, 0);

  346.                        

  347.                     }

  348.         }

  349.  

  350.         return handled || super.onGenericMotionEvent(event);

  351.     }
