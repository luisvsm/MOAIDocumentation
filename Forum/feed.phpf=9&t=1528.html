Code:
  1. -- Pause

  2. oldRoot = MOAIActionMgr.getRoot ()

  3. MOAIActionMgr.setRoot ()

  4.  

  5. local thread = MOAIThread.new ()

  6. thread:run ( doPaused )

  7.  

  8. -- doPaused

  9. function doPaused ()

  10.   MOAIInputMgr.device.touch:setCallback( onTouchEvent )

  11. end

  12.  

  13. -- Resume

  14. MOAIActionMgr.setRoot ( oldRoot )
