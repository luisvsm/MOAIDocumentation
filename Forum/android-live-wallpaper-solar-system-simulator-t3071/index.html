With very few changes, a MoaiView can serve as the GLSurfaceView for a WallpaperService.Engine subclass.

The WallpaperService.Engine subclass just overrides/implements a handful of methods. And the WallpaperService subclass, other than implementing onCreateEngine method, it just does some things that the MoaiActivity class does. It needs to do these things because there is no activity, only a service.

I'm no Java guru, and I know some parts of this code are unnecessary. But I'll just dump my working code here, in case I get run over by a bus.

  2.  

  3. import android.annotation.TargetApi;

  4. import android.content.Context;

  5. import android.content.SharedPreferences;

  6. import android.content.pm.PackageManager;

  7. import android.os.Build;

  9. import android.view.SurfaceHolder;

  12. import com.ziplinegames.moai.MoaiLog;

  13.  

  14. import java.util.Map;

  16. /**

  17.  * An example Live Wallpaper service for Moai.

  18.  *

  19.  */

  20. public class MoaiWallpaperService extends WallpaperService {

  21.     private static final Object lock = new Object();

  22.     MyWallpaperEngine mEngine;

  23.     public static final String strSharedPrefs="BGSettings";

  24.     public static String sDocumentsDirectory;

  25.  

  26.     // borrowed from MoaiActivity

  27.     static {

  28.         MoaiLog.i("Loading libmoai.so");

  29.         System.loadLibrary("moai");

  30.     }

  31.  

  32.     @Override

  33.     public Engine onCreateEngine() {

  34.         MoaiLog.i("MoaiWallpaperService.onCreateEngine");

  35.         // do some things MoaiActivity would do...

  36.         Moai.onCreate(this);

  37.         Moai.createContext();

  38.         Moai.init();

  39.  

  40.         android.content.pm.PackageManager m = getPackageManager();

  41.         String s = getPackageName();

  42.         try {

  43.             android.content.pm.PackageInfo p = m.getPackageInfo(s, 0);

  44.             Moai.mount("bundle", p.applicationInfo.sourceDir);

  45.             Moai.setWorkingDirectory("bundle/assets/lua");

  46.         }

  47.         catch (PackageManager.NameNotFoundException e) {

  48.             MoaiLog.w("Error: Package name not found.");

  49.         }

  50.         sDocumentsDirectory = getFilesDir().getAbsolutePath();

  51.         Moai.setDocumentDirectory ( getFilesDir ().getAbsolutePath ());

  52.  

  53.         //if (mEngine != null)

  54.         //    mEngine.onVisibilityChanged(false);

  55.  

  56.         mEngine = new MyWallpaperEngine(this);

  57.         mEngine.setTouchEventsEnabled(true);

  58.  

  59.  

  60.         return mEngine;

  61.     }

  62.  

  63.     @Override

  64.     public void onDestroy() {

  65.         MoaiLog.i("MoaiWallpaperService.onDestroy");

  66.         super.onDestroy();

  67.     }

  68.  

  69.     // Based on: <!-- m --><a class="postlink" href="http://www.learnopengles.com/how-to-use-opengl-es-2-in-an-android-live-wallpaper/">http://www.learnopengles.com/how-to-use ... wallpaper/</a><!-- m -->

  70.     protected class MyWallpaperEngine extends Engine {

  71.         protected MyGLSurfaceView mGLSurfaceView = null;

  72.         protected boolean mVisible = true;

  73.         MoaiWallpaperService mContext = null;

  74.  

  75.         protected MyWallpaperEngine(MoaiWallpaperService context) {

  76.             super();

  77.             MoaiLog.i("MyWallpaperEngine.MyWallpaperEngine");

  78.             mContext = context;

  79.         }

  80.  

  81.         @TargetApi(Build.VERSION_CODES.GINGERBREAD)

  82.         @Override

  83.         public void onTouchEvent(android.view.MotionEvent event) {

  84.             if (mGLSurfaceView == null)

  85.                 return;

  86.             synchronized (lock) {

  87.                 super.onTouchEvent(event);

  88.                 mGLSurfaceView.onTouchEvent(event);

  89.             }

  90.         }

  91.  

  92.         @Override

  93.         public void onVisibilityChanged(boolean visible) {

  94.             //mVisible = visible;

  95.  

  96.             if (mGLSurfaceView == null)

  97.                 return;

  98.             //mGLSurfaceView.mIsVisible = visible;

  99.             synchronized (lock) {

  100.                 super.onVisibilityChanged(visible);

  101.                 if (visible) {

  102.                     //mGLSurfaceView.pause(false);

  103.                     MoaiLog.i("onPause: visible: true");

  104.                     mGLSurfaceView.onResume();

  105.                     Moai.onResume();

  106.                     Moai.pause(false);

  107.                 }

  108.                 else {

  109.                     //mGLSurfaceView.pause(true);

  110.                     MoaiLog.i("onPause: visible: false");

  111.                     mGLSurfaceView.onPause();

  112.                     Moai.onPause();

  113.                     Moai.pause(true);

  114.                 }

  115.             }

  116.         }

  117.  

  118.         @Override

  119.         public void onSurfaceChanged(android.view.SurfaceHolder holder, int format, int width, int height) {

  120.             if (mGLSurfaceView == null)

  121.                 return;

  122.             synchronized (lock) {

  123.                 super.onSurfaceChanged(holder, format, width, height);

  124.                 mGLSurfaceView.surfaceChanged(holder, format, width, height);

  125.             }

  126.         }

  127.  

  128.         @Override

  129.         public void onSurfaceCreated(android.view.SurfaceHolder holder) {

  130.             //if (!isPreview())

  131.             //    mGLSurfaceView = new MyGLSurfaceView(mContext);

  132.             if (mGLSurfaceView == null || !mVisible)

  133.                 return;

  134.             synchronized (lock) {

  135.                 MoaiLog.i("MyWallpaperEngine.onSurfaceCreated");

  136.                 super.onSurfaceCreated(holder);

  137.                 mGLSurfaceView.surfaceCreated(holder);

  138.             }

  139.         }

  140.  

  141.         @Override

  142.         public void onCreate(android.view.SurfaceHolder surfaceHolder) {

  143.             synchronized (lock) {

  144.                 MoaiLog.i("MyWallpaperEngine.onCreate");

  145.                 mGLSurfaceView = new MyGLSurfaceView(mContext);

  146.                 super.onCreate(surfaceHolder);

  147.             }

  148.         }

  149.  

  150.         @Override

  151.         public void onDestroy() {

  152.             //if (mGLSurfaceView == null)

  153.             //    return;

  154.             //mVisible = false;

  155.             synchronized (lock) {

  156.                 MoaiLog.i("MyEngine ONDESTROY SOFTLY");

  157.                 //super.onDestroy();

  158.                 //mGLSurfaceView.onDestroy();

  159.                 //mGLSurfaceView.onPause();

  160.                 //mGLSurfaceView.pause(true);

  161.                 //mGLSurfaceView = null;

  162.             }

  163.         }

  164.  

  165.         protected class MyGLSurfaceView extends MoaiView {

  166.             protected MyGLSurfaceView(MoaiWallpaperService context) {

  167.                 // This width and height is only temporary, they

  168.                 // will be updated by superclass onSurfaceChanged.

  169.                 //

  170.                 // You can get the current width and height

  171.                 // from within your Moai Lua application

  172.                 // by calling MOAIGfxDevice.getViewSize()

  173.                 super(context, 1920, 1200, 0x00020000);

  174.             }

  175.  

  176.             @Override

  177.             public SurfaceHolder getHolder() {

  178.                 return getSurfaceHolder();

  179.             }

  180.         }

  181.     }

  182. }
