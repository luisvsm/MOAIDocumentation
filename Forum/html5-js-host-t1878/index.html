Not sure if I helped it any but I did something like this:

Code: Select all
  1. Module.postRun.push(function() {

  2.             console.log("post run");

  3.             moaijs.runhost();

  4.         });

  5.         setTimeout(function() {

  6.             Module.run();

  7.         }, 1);



So, a timeout to let Module do whatever in it's thread before we run, and then the post run to make sure the module is up and running. Might be a bit too much but it seems to work in both Chrome and FF so far.

Here's my mods to the host.js
Code: Select all
  1. moaijs.run = function() {

  2.     console.log("moai host run");

  3.     Module = (typeof Module === "undefined")?{}:Module;

  4.     Module['canvas'] = document.getElementById('MoaiCanvas');

  5.     Module['setStatus'] = function(status) {

  6.         console.log(status);

  7.         document.getElementById('MoaiStatus').innerHTML = status;

  8.     };

  9.     Module['printErr'] = function(text) {

  10.         text = Array.prototype.slice.call(arguments).join(' ');

  11.         console.log('Error:'+text);

  12.     }

  13.     Module['noExitRuntime'] = true;

  14.     Module['totalDependencies'] = 0;

  15.  

  16.     Module['monitorRunDependencies'] = function(left) {

  17.         this.totalDependencies = Math.max(this.totalDependencies, left);

  18.         Module.setStatus(left ? 'Preparing... (' + (this.totalDependencies-left) + '/' + this.totalDependencies + ')' : 'All downloads complete.');

  19.         if (!left) {

  20.             moaijs.startFromModule();

  21.         }

  22.     }

  23.     $(document).ready(moaijs.startFromBody);

  24. }

  25.  

  26. moaijs.race = {

  27.     module: false,

  28.     body: false

  29. }

  30. moaijs.startFromModule = function() {

  31.     console.log("Module ready");

  32.     moaijs.race.module = true;

  33.     moaijs.tryStart();

  34. }

  35. moaijs.startFromBody = function() {

  36.     console.log("Page ready");

  37.     moaijs.race.body = true;

  38.     moaijs.tryStart();

  39. }

  40. moaijs.tryStart = function() {

  41.     if (moaijs.race.module && moaijs.race.body) {

  42.         Module.postRun.push(function() {

  43.             console.log("post run");

  44.             moaijs.runhost();

  45.         });

  46.         setTimeout(function() {

  47.             Module.run();

  48.         }, 1);

  49.     }

  50. }
