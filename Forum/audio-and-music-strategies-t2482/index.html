I've come up with a simple solution to play a sound more than once. Build a pool of sounds and if the same sound is attempted to play twice, then use a sound from the pool instead.

Code: Select all
  1. Audio.pool_size = 3

  2.  

  3. function Audio:BuildPool()

  4.     Audio.pool = {}

  5.     for i=1,Audio.pool_size do

  6.         local sound = MOAIUntzSound.new()

  7.         sound.name = false

  8.         Audio.pool&#91;i&#93; = sound

  9.     end

  10. end

  11.  

  12. function Audio:PlaySound(name, prop, force)

  13.         if not Audio.sound_on then return end

  14.         if Audio.sounds&#91;name&#93; then

  15.         if not Audio.sounds&#91;name&#93;:isPlaying() or force then

  16.             Audio.sounds&#91;name&#93;:setVolume(Audio.sound_volume*Audio.max_volume)

  17.             Audio.sounds&#91;name&#93;:play()

  18.         else

  19.             -- play twice, but not overlapping

  20.             if Audio.sounds&#91;name&#93;:getPosition() > 0.1 then

  21.                 Audio:PlayPool(name)

  22.             end

  23.         end

  24.     else

  25.         print('missing sound: '..name)

  26.     end

  27. end

  28.  

  29. function Audio:PlayPool(name)

  30.     local played = false

  31.     for i=1, Audio.pool_size do

  32.         local sound = Audio.pool&#91;i&#93;

  33.         if sound.name == name then

  34.             if not sound:isPlaying() then

  35.                 sound:setVolume(Audio.sound_volume*Audio.max_volume)

  36.                 sound:play()

  37.                 return

  38.             end

  39.         end

  40.     end

  41.     for i=1, Audio.pool_size do

  42.         local sound = Audio.pool&#91;i&#93;

  43.         if not sound:isPlaying() then

  44.             local file = Audio:GetSoundFilename(name)

  45.             if MOAIFileSystem.checkFileExists(file) then

  46.                 sound.name = name

  47.                 sound:setVolume(Audio.sound_volume*Audio.max_volume)

  48.                 sound:load(file)

  49.                 sound:play()

  50.             end

  51.             return

  52.         end

  53.     end

  54. end



However I don't think this is ultimately the best solution since it is not taking true consideration of the polyphony of the system. So you could play way more than 4 or 8 sounds or whatever is the system limit. I suppose you could look through all your sounds and count how many are playing before playing another one. So while this code is not ideal, might as well share it before scrapping it for something better.

Also there should be a dedicated "channel" for the player so that their sound is always played and not pushed out by other sounds.

  71.                 volume = 1

  74.                 volume = 1

  92.         end

  93.

  95.                 loop = false

  98.                 return nil

  101.                 return nil

  102.         end

  104.                 return nil

  316. -- immediately change the volume of a sound

  317. function MyMAudio.setVolume(params)

  318.         local soundId = params.soundId

  319.         local volume = params.volume

  320.         local sound = MyMAudio.getSoundNode(soundId)

  321.         if (sound ~= nil) then

  322.                 local baseVolume = volume

  323.                

  324.                 if (sound.channelType == CHANNEL_MUSIC) then

  325.                         volume = volume * masterMusicVolume

  326.                 elseif (channelType == CHANNEL_AMBIENT) then

  327.                         volume = volume * masterAmbientVolume

  328.                 else

  329.                         volume = volume * masterSoundVolume

  330.                 end

  331.                

  332.                 sound.baseVolume = baseVolume

  333.                 sound.sound:setVolume(volume)

  334.         end

  335. end

  339. -- params.newVolume : new the new volume

  341. function MyMAudio.changeVolume(params)

  342.         local soundId = params.soundId

  343.         local newVolume = params.newVolume

  344.         local time = params.time

  345.         if (not time) then

  346.                 time = 5

  348.         local sound = MyMAudio.getSoundNode(soundId)

  349.         if (sound ~= nil) then

  350.                 local baseVolume = volume

  351.                 if (sound.channelType == CHANNEL_MUSIC) then

  352.                         newVolume = newVolume * masterMusicVolume

  353.                 elseif (channelType == CHANNEL_AMBIENT) then

  354.                         newVolume = newVolume * masterAmbientVolume

  355.                 else

  356.                         newVolume = newVolume * masterSoundVolume

  357.                 end

  358.                

  359.                 sound.baseVolume = baseVolume

  360.                 sound.sound:seekVolume(newVolume, time)

  361.         end

  544.         if (not MyMAudio.isMusicConfigured()) then

  545.                 return

  546.         end

  547.        

  548.         if (MYMMOAIAudioAndroid.isMusicPlayerExist(soundFile)) then

  551.                 --end

  552.         else

  553.                 local c = channel

  554.                 local mv = MyMAudio.getMaxMusicVolume() * masterMusicVolume

  555.                 if (c == CHANNEL_AMBIENT) then

  556.                         c = 1

  557.                         mv = masterAmbientVolume

  558.                 else

  559.                         c = 0

  560.                 end

  562.         end

  563.        

  564.         if (params and params.fadeIn) then

  565.                 local time = params.time or 2

  566.                 time = time * 0.1

  567.                 MyMAudio.fadeInMusicForAndroid(channel, soundFile, 0, time)
