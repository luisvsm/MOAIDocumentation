It isn't explicitly documented right now. Once again, this will be added to the API docs, but I only have two hands...

Meanwhile:

Code: Select all
  1. function GetStaticStatus(T)

  2.         local static = false;

  3.        

  4.         if type(T) == "table" then

  5.                 for n in pairs(T) do

  6.                         if n == 'new' then

  7.                                 return "new()"

  8.                         end                        

  9.                 end

  10.                 static = true;

  11.         end

  12.         mt = getmetatable(T)

  13.        

  14.         if mt and type(mt.__index)=="table" then

  15.                 for n,m in pairs (mt.__index) do

  16.                         if n == 'new' then return "new()" end

  17.                 end

  18.                 static = true

  19.         elseif mt and type(mt.__index)=="function" then

  20.                 return "?? - __index is a function."

  21.         end

  22.  

  23.         if static then

  24.                 return "static"

  25.         else

  26.                 return "# Not a class."

  27.         end

  28. end

  29.  

  30.  

  31. acc= {}

  32. for k,v in pairs(_G) do

  33.     if k:find"MOAI" then table.insert(acc,k) end

  34. end

  35. table.sort(acc)

  36. for _,k in ipairs(acc) do

  37.         print (k, GetStaticStatus(_G[k]))

  38. end

  39.
