My code is similar. It looks like this:

Code:
  1. ----------------------------------------

  2. -- in main.lua

  3.  

  4. -- Main Loop

  5. function mainLoop ( pState )

  6.   MOAIUntzSystem.initialize ( )

  7.   -- game loop here

  8. end

  9.  

  10. ----------------------------------------

  11. -- Sound.lua

  12. module ( ..., package.seeall )

  13.  

  14. local bgm

  15.  

  16. function playSound(pFileName, pLoadIntoMemory, pVolume, pLoop)

  17.   local result = MOAIUntzSound.new ()

  18.   result:load( pFileName, pLoadIntoMemory )

  19.   result:setVolume( pVolume )

  20.   result:setLooping( pLoop )

  21.   result:play()

  22.   return result

  23. end

  24.  

  25. function playSFX(pFileName)

  26.   local loadIntoMemory = false

  27.   local volume         = 2

  28.   local loop           = false

  29.   sfx = playSound(pFileName, loadIntoMemory, volume, loop)

  30.   return sfx

  31. end

  32.  

  33. function playBGM(pFileName)

  34.   local loadIntoMemory = false

  35.   local volume         = 0.5

  36.   local loop           = true

  37.   if bgm then

  38.     bgm:stop()

  39.     bgm = nil

  40.   end

  41.   bgm = playSound(pFileName, loadIntoMemory, volume, loop)

  42.   return bgm

  43. end
